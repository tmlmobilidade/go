1 - Get validation 
2 - setDate cutoffDate = 30 days in unixtimestamp - done
3 - Loop on each validation
    3.1 - For this validation delete on created_at




import LOGGER from '@helperkits/logger';
import TIMETRACKER from '@helperkits/timer';
import { validations } from '@tmlmobilidade/interfaces';

async function cleanOldValidations() {
 try {
   
   LOGGER.init();
   const globalTimer = new TIMETRACKER();
  
   const validationsCollection = await validations.getCollection();


   const thirtyDaysInMs = 30 * 24 * 60 * 60 * 1000;
   const cutoffTimestamp = Date.now() - thirtyDaysInMs;
   
   
   const oldValidations = await validationsCollection.find({
     created_at: { $lt: cutoffTimestamp }

   }).toArray();

   LOGGER.info(`Found ${oldValidations.length} validations older than 30 days`);
 

   if (oldValidations.length > 0) {
     const deleteTimer = new TIMETRACKER();
     let deletedCount = 0;

     // 3.1 Loop through each validation
     for (const validation of oldValidations) {
       try {

         await validationsCollection.deleteOne({ _id: validation._id });
         deletedCount++;
       } catch (err) {

         LOGGER.error(`Error deleting validation ${validation._id}:`, err);
       }
     }
     LOGGER.info(`Deleted ${deletedCount} validations. (${deleteTimer.get()})`);
   } else {

     LOGGER.info('No old validations found to clean');
   }
   LOGGER.terminate(`Cleanup completed in ${globalTimer.get()}`);
 } catch (err) {

   LOGGER.error('An error occurred while cleaning old validations:', err);

   LOGGER.error('Retrying in 10 seconds...');
   
   setTimeout(() => {
     process.exit(0);
   }, 10000);
 }
}
